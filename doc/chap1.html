1 Loop integrals

1.1 Introduction

This package is an ongoing work to compute master integrals using commutative and noncommutative methods from computational algebraic geometry.

1.2 Example

gap> LOOP_INTEGRALS.Dimension := 4;
4
gap> LD := LoopDiagram( "l1..2", "p" );
<A loop diagram with loop momenta [ l1, l2 ] & external momenta [ p ]>
gap> SetPropagators( LD, [ l1^2, (l1+p)^2, l2^2, (l2-p)^2, (l1+l2)^2 ] );
gap> SetNumerators( LD, [ ] );
gap> SetRelationsOfMomenta( LD, [ ] );
gap> SetIndependetLorentzInvariants( LD, [ l1^2, l2^2, l1*l2, l1*p, l2*p, p^2 ] );
gap> SetExtraLorentzInvariants( LD, [ p^2 ] );
gap> E12 := PairOfMatricesOfLoopDiagramInPropagators( LD );
[ <A non-zero 6 x 5 matrix over an external ring>,
  <A non-zero 5 x 5 matrix over an external ring> ]
gap> Display( E12[1] );
2*D1,     D1+D2-x6,    0,        0,           D1-D3+D5,
-D1-D3+D5,-D1-D4+D5+x6,0,        0,           -D1+D3+D5,
-D1+D2-x6,-D1+D2+x6,   0,        0,           -D1+D2+D3-D4,
0,        0,           -D1-D3+D5,-D2-D3+D5+x6,D1-D3+D5,
0,        0,           2*D3,     D3+D4-x6,    -D1+D3+D5,
0,        0,           D3-D4+x6, D3-D4-x6,    -D1+D2+D3-D4
gap> Display( E12[2] );
D1,0, 0, 0, 0,
0, D2,0, 0, 0,
0, 0, D3,0, 0,
0, 0, 0, D4,0,
0, 0, 0, 0, D5
gap> S := SyzygiesOfRows( E12[1], E12[2] );
<A non-zero 19 x 6 matrix over an external ring>
gap> Display( EntriesOfHomalgMatrix( CertainRows( S, [ 1 ] ) ) );
[ 2*D2, 0, 0, D3+D4-x6, D1+D2+2*D4-2*D5-x6, -D1+D3+D5 ]
gap> Display( EntriesOfHomalgMatrix( CertainRows( S, [ 2 ] ) ) );
[ 2*D2+D3+D4-2*D5-x6, D1+D2-x6, -D1+D3-D5, 0, 2*D4, 0 ]
gap> Display( EntriesOfHomalgMatrix( CertainRows( S, [ 3 ] ) ) );
[ D1-D2, 0, D1, 0, D3-D4, -D3 ]
gap> Display( EntriesOfHomalgMatrix( CertainRows( S, [ 4 ] ) ) );
[ 0, 0, 0, 0, D4*D5, 0 ]

gap> LOOP_INTEGRALS.Dimension := 2;
2
gap> LD := LoopDiagram( "l1", "k1..2,k4" );
<A loop diagram with loop momenta [ l1 ] & external momenta [ k1, k2, k4 ]>
gap> SetPropagators( LD, [ l1^2, (l1-k1)^2, (l1-k1-k2)^2, (l1+k4)^2 ] );
gap> SetNumerators( LD, [ ] );
gap> rel1 := List( ExternalMomenta( LD ), k -> k^2 );;
gap> rel2 := [ (k1+k2+k4)^2 ];;
gap> SetRelationsOfMomenta( LD, Concatenation( rel1, rel2 ) );
gap> SetIndependetLorentzInvariants( LD,
>         [ l1^2, l1*k1, l1*k2, l1*k4, 2*k1*k2, 2*k1*k4 ] );
gap> SetExtraLorentzInvariants( LD, [ 2*k1*k2, 2*k1*k4 ] );
gap> E12 := PairOfMatricesOfLoopDiagramInPropagators( LD );
[ <A non-zero 4 x 4 matrix over an external ring>,
  <A non-zero 4 x 4 matrix over an external ring> ]
gap> Display( E12[1] );
2*D1,    D1+D2,    D1+D3-x5, D1+D4,
D1-D2,   D1-D2,    D1-D2-x5, D1-D2+x6,
D2-D3+x5,D2-D3,    D2-D3,    D2-D3-x6,
-D1+D4,  -D1+D4-x6,-D1+D4+x5,-D1+D4
gap> S := SyzygiesOfRows( E12[1], E12[2] );
<A non-zero 8 x 4 matrix over an external ring>
gap> Sred := ReducedBasisOfRowModule( S );
<A non-zero 6 x 4 matrix over an external ring>
gap> Display( EntriesOfHomalgMatrix( CertainRows( Sred, [ 1 ] ) ) );
[ D2-D4, D4, 0, D2 ]
gap> Display( EntriesOfHomalgMatrix( CertainRows( Sred, [ 2 ] ) ) );
[ D1-D3, -D1, -D1, 0 ]
gap> Display( EntriesOfHomalgMatrix( CertainRows( Sred, [ 3 ] ) ) );
[ 2*D3*D4-2*D4^2-D4*x5, D2*D4-D3*D4+2*D4^2+D4*x5, D1*D4+D2*D4, 2*D2*D4 ]

gap> LOOP_INTEGRALS.Dimension := 2;
2

>           (l1-l2)^2, (l1-l2+p2)^2 ] );
gap> SetNumerators( LD, [ l1^2 ] );
gap> rel := List( ExternalMomenta( LD ), p -> p^2 );;
gap> SetRelationsOfMomenta( LD, rel );
gap> SetIndependetLorentzInvariants( LD,
>         [ l1^2, l2^2, l1*l2,
>           l1*p1, l2*p1, l1*p2, l2*p2,
>           p1*p2 ] );
gap> SetExtraLorentzInvariants( LD, [ p1*p2 ] );
gap> E12 := PairOfMatricesOfLoopDiagramInPropagators( LD );
[ <A non-zero 8 x 6 matrix over an external ring>,
  <A non-zero 6 x 6 matrix over an external ring> ]
gap> Display( E12[1] );
D1+N7,     D2+N7,      0,          0,       -D3+D5+N7,   D2-D3+D5,

-D1+N7,    -D1+N7+2*x8,0,          0,       -D1-D3+D4+N7,_[3,6],
D2-N7-2*x8,D2-N7,      0,          0,       -D5+D6,      -D5+D6,
0,         0,          D3-D5+N7,   D1+D3-D5,D3-D5-N7,    -D2+D3-D5,
0,         0,          2*D3,       D3+D4,   D3+D5-N7,    -D2+D3+D6,
0,         0,          D3-D4,      D3-D4,   D1+D3-D4-N7, _[7,6],
0,         0,          D2+D5-D6-N7,_[8,4],  D5-D6,       D5-D6
gap> S := SyzygiesOfRows( E12[1], E12[2] );
<A non-zero 52 x 8 matrix over an external ring>
